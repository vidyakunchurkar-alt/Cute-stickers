    <script>
        // Copy emoji to clipboard
        function copyEmoji(emoji) {
            navigator.clipboard.writeText(emoji).then(() => {
                createHeart(emoji);
                // Show copy feedback
                const btn = event.target;
                const originalText = btn.textContent;
                btn.textContent = 'Copied! ✨';
                btn.style.background = '#4ecdc4';
                setTimeout(() => {
                    btn.textContent = originalText;
                    btn.style.background = '';
                }, 1500);
            });
        }

        // Create floating hearts
        function createHeart(emoji) {
            const heart = document.createElement('div');
            heart.className = 'heart';
            heart.innerHTML = emoji || '💖';
            heart.style.left = Math.random() * 100 + '%';
            heart.style.animationDuration = (Math.random() * 3 + 4) + 's';
            document.getElementById('hearts').appendChild(heart);

            setTimeout(() => {
                heart.remove();
            }, 7000);
        }

        // Auto-generate floating hearts
        setInterval(() => {
            if (Math.random() > 0.7) {
                createHeart();
            }
        }, 3000);

        // Add click animation to cards
        document.querySelectorAll('.sticker-card').forEach(card => {
            card.addEventListener('click', function(e) {
                if (!e.target.classList.contains('download-btn')) {
                    const rect = this.getBoundingClientRect();
                    const x = e.clientX - rect.left;
                    const y = e.clientY - rect.top;
                    createRipple(x, y, this);
                }
            });
        });

        function createRipple(x, y, element) {
            const ripple = document.createElement('div');
            ripple.style.position = 'absolute';
            ripple.style.borderRadius = '50%';
            ripple.style.background = 'rgba(255, 107, 157, 0.6)';
            ripple.style.width = '20px';
            ripple.style.height = '20px';
            ripple.style.left = x - 10 + 'px';
            ripple.style.top = y - 10 + 'px';
            ripple.style.transform = 'scale(0)';
            ripple.style.animation = 'ripple 0.6s linear';
            element.style.overflow = 'hidden';
            element.appendChild(ripple);

            setTimeout(() => {
                ripple.remove();
            }, 600);
        }

        // Add ripple animation
        const style = document.createElement('style');
        style.textContent = `
            @keyframes ripple {
                to {
                    transform: scale(4);
                    opacity: 0;
                }
            }
        `;
        document.head.appendChild(style);
    </script>
